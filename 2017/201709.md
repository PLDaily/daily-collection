### 二

> 判断是否为Undefined

```
var isUndefined = function(obj) {
	return obj === void 0;
};
```

> 判读是否为数组

```
var isArray = nativeIsArray || function(obj) {
    return toString.call(obj) === '[object Array]';
};
```

> 判断是否为对象

```
var isObject = function(obj) {
    var type = typeof obj;
    return type === 'function' || type === 'object' && !!obj;
};
```
